// ---------------- DEVICE JSON ----------------
String deviceJson(const char* name, const char* id_suffix) {
    return String("{") +
           "\"identifiers\":[\"" DEVICE_ID "_" + String(id_suffix) + "\"]," +
           "\"name\":\"" + String(name) + "\"," +
           "\"model\":\"" DEVICE_MODEL "\"," +
           "\"manufacturer\":\"" DEVICE_MANUFACTURER "\"" +
           "}";
}

// ---------------- HA INIT ----------------
template <typename T>
void haInit(RoboEyes<T>* eyes) {
    roboEyes = eyes;
    publishTouchDiscovery();
    mqttPublish(STATUS_TOPIC, "online", true); // retain status
}

// Explicit instantiation
template void haInit(RoboEyes<Adafruit_SH1106G>* eyes);

// ---------------- SEND TOUCH EVENT ----------------
void haPublishTouch(const char* event) {
    String topic = String(TOUCH_TOPIC) + "/" + String(event); // single/double/long
    mqttPublish(topic.c_str(), "pressed", false);
}
